🧠 Recommendation Logic

We'll compute:

  • Similarity to a given song using features (title, artist, genre, etc.)

  • Context relevance (e.g., match genre = 1.0, duration difference = small = better)

  • Combine them using a weighted score:
    final_score = α * similarity + β * context_score

You can tune α and β to favor similarity or context more.

import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.preprocessing import MinMaxScaler
from sklearn.metrics.pairwise import cosine_similarity

def build_song_features(df):
    # Combine relevant features into a string
    df['combined'] = df['artist'] + ' ' + df['album'] + ' ' + df['genre'] + ' ' + df['name']
    
    # TF-IDF vector for combined textual features
    tfidf = TfidfVectorizer()
    tfidf_matrix = tfidf.fit_transform(df['combined'])
    
    # Normalize duration
    scaler = MinMaxScaler()
    duration_scaled = scaler.fit_transform(df[['duration']])
    
    return tfidf_matrix, duration_scaled

def recommend_songs(df, song_index, genre=None, duration=None, alpha=0.7, beta=0.3, top_n=10):
    tfidf_matrix, duration_scaled = build_song_features(df)
    
    # Content similarity
    sim_scores = cosine_similarity(tfidf_matrix[song_index], tfidf_matrix).flatten()
    
    # Context score
    context_scores = []
    for i, row in df.iterrows():
        genre_score = 1.0 if genre and row['genre'].lower() == genre.lower() else 0.0
        duration_score = 1 - abs(duration_scaled[song_index][0] - duration_scaled[i][0]) if duration else 0.0
        context_scores.append((genre_score + duration_score) / 2)
    context_scores = pd.Series(context_scores)

    # Final score
    final_scores = alpha * sim_scores + beta * context_scores
    recommended_indices = final_scores.argsort()[::-1][1:top_n+1]
    
    return df.iloc[recommended_indices][['name', 'artist', 'genre', 'duration']]
